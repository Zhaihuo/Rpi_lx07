cmake_minimum_required(VERSION 3.16)

project(E29_EOLTest VERSION 0.1 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD_REQUIRED ON)

find_package(Qt6 6.5 REQUIRED COMPONENTS Quick QuickControls2 Widgets)

qt_standard_project_setup(REQUIRES 6.5)

qt_add_executable(appE29_EOLTest
    main.cpp
    # uart.cpp
)

qt_add_qml_module(appE29_EOLTest
    URI E29_EOLTest
    VERSION 1.0
    QML_FILES Main.qml
    QML_FILES DefaultPage.qml
    SOURCES cppdata.h cppdata.cpp uart.cpp
    RESOURCES images/test001.png images/test002.png images/test003.png images/test004.png images/test005.png images/test006.png images/test007.png images/test008.png
    RESOURCES images/test009.png images/test010.png images/test011.png images/test012.png images/test013.png images/test014.png images/test015.png
)

# Qt for iOS sets MACOSX_BUNDLE_GUI_IDENTIFIER automatically since Qt 6.1.
# If you are developing for iOS or macOS you should consider setting an
# explicit, fixed bundle identifier manually though.
set_target_properties(appE29_EOLTest PROPERTIES
#    MACOSX_BUNDLE_GUI_IDENTIFIER com.example.appTouchAreaTest
    MACOSX_BUNDLE_BUNDLE_VERSION ${PROJECT_VERSION}
    MACOSX_BUNDLE_SHORT_VERSION_STRING ${PROJECT_VERSION_MAJOR}.${PROJECT_VERSION_MINOR}
    MACOSX_BUNDLE TRUE
    WIN32_EXECUTABLE TRUE
)

target_link_libraries(appE29_EOLTest
    PRIVATE Qt6::Quick
    Qt6::QuickControls2
    Qt6::Widgets
)

include(GNUInstallDirs)
install(TARGETS appE29_EOLTest
    BUNDLE DESTINATION .
    LIBRARY DESTINATION ${CMAKE_INSTALL_LIBDIR}
    RUNTIME DESTINATION ${CMAKE_INSTALL_BINDIR}
)
